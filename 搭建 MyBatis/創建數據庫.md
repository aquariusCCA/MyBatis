```sql
create database mybatis;
use mybatis;

DROP TABLE IF EXISTS t_user;
CREATE TABLE t_user (
  eid       INT PRIMARY KEY AUTO_INCREMENT,
  emp_name  VARCHAR(20),
  age       INT,
  sex       TINYINT UNSIGNED NOT NULL COMMENT '1=男, 2=女',
  email     VARCHAR(50),
  did       INT,
  CONSTRAINT chk_sex CHECK (sex IN (1, 2)) -- 需 MySQL ≥ 8.0.16
) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;


DROP TABLE if exists t_dept;
create table t_dept(
    did INT PRIMARY KEY AUTO_INCREMENT,
    dept_name VARCHAR(20)
);
```

#  t_user 添加 password 欄位

```sql
DROP TABLE IF EXISTS t_user;
CREATE TABLE t_user (
	eid       INT PRIMARY KEY AUTO_INCREMENT,
	emp_name  VARCHAR(20),
	password  VARCHAR(20),
	age       INT,
	sex       TINYINT UNSIGNED NOT NULL COMMENT '1=男, 2=女',
	email     VARCHAR(50),
	did       INT,
	CONSTRAINT chk_sex CHECK (sex IN (1, 2)) -- 需 MySQL ≥ 8.0.16
) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
```
