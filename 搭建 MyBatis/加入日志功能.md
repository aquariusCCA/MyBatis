# 加入依赖

```xml
<!-- Logback -->  
<dependency>  
    <groupId>ch.qos.logback</groupId>  
    <artifactId>logback-classic</artifactId>  
    <version>1.5.6</version>  
</dependency>
```

---

# mybatis 日志输出配置

```xml
<settings>  
    <!-- SLF4J -->  
    <setting name="logImpl" value="SLF4J"/>  
</settings>
```

---

# 加入 logback 的配置文件

- 配置文件名为 logback.xml，存放的位置是 src/main/resources 目录下

```xml
<configuration>
    <appender name="STDOUT" class="ch.qos.logback.core.ConsoleAppender">
        <encoder>
            <pattern>%d{HH:mm:ss.SSS} %-5level %logger{36} - %msg%n</pattern>
        </encoder>
    </appender>

    <!-- Mapper 命名空間/套件：開 DEBUG 才會看到 ==> Preparing/Parameters -->
    <logger name="com.atguigu.mybatis.mapper" level="DEBUG" additivity="false">
        <appender-ref ref="STDOUT"/>
    </logger>

    <!-- MyBatis 相關 logger：DEBUG 才會列出 SQL/參數/結果
    <logger name="org.apache.ibatis" level="DEBUG" additivity="false">
        <appender-ref ref="STDOUT"/>
    </logger>
    -->

    <!-- 有些驅動/連線池會把 JDBC 層細節也打出來；需要時再打開
    <logger name="java.sql" level="DEBUG" additivity="false">
      <appender-ref ref="STDOUT"/>
    </logger>
    -->

    <!-- 其他框架/應用預設 INFO，降低雜訊 -->
    <root level="INFO">
        <appender-ref ref="STDOUT"/>
    </root>
</configuration>
```
